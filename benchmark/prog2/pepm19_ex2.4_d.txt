%PROGRAM
let fin = event fin; ()

let repeat f x =
  if x <= 0 then fin
  else (
    if<d1> * then f x (repeat f)
    else repeat f (x - 1)
  )

let () =
  let y = <a1> * in
  /* この条件を無くしてdemonicにするとinvalidになる */
  repeat (\x k. k (x - y)) 3

/*
   phi * *
-> Input 0 (\x. ここはそのままで、unit型.単にnondetintが変わっただけ phi * *)
*/

%TRANSITION
q0 d1 -> (1, q0) /\ (2, q0).
q0 a1 -> (1, q0) \/ (2, q1).
q0 fin -> true.

q1 d1 -> false.
q1 a1 -> (1, q0) \/ (2, q1).
q1 fin -> false.

%PRIORITY
q0 -> 1.
/* demonicのときは偶数、angelicのときは奇数の大きな値にする */
q1 -> 3.
