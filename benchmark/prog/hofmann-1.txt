Example 1 of Hofmann and Chen, "Buchi Types for Infinite Traces and Liveness", CSL-LICS 2014

Trace(P) \subset L(A)

%HES
let f = event a; event b; f
let () = f

%INITIAL
qa

%TRANSITION
(qa, a) -> qa
(qb, a) -> qa
(qa, b) -> qb
(qb, b) -> qb

%PRIORITY
qa -> 1
qb -> 2
