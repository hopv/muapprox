%PROGRAM
let check fx k =
  fx (\x. if x > 0 then k 1 else k 0)

let pred fx k =
  fx (\x.
    k (x - 1)
  )

let g k = if * then (event a; k) else (event b; k)

let f fx =
  check fx (\b.
    if b = 1 then g (f (pred fx)) else (event b; f (\k. k 1))
  )

let () =
  let x = * in
  f (\k. k x)

%TRANSITION
qa a -> qa
qa b -> qb

qb a -> qa
qb b -> qb

%PRIORITY
qa -> 0
qb -> 1
