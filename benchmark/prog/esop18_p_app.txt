%PROGRAM
let fb x = if x > 0 then (event a; app fa (x-1)) else (event b; app fb 5)
let fa x = if x > 0 then (event a; app fa (x-1)) else (event b; app fb 5)
let app h x = h x
let () = fb 5

%TRANSITION
qa a -> qa
qa b -> qb

qb a -> qa
qb b -> qb

%PRIORITY
qa -> 0
qb -> 1

/*
## プログラムの性質
a, bともに、必ず無限回現れる

## オートマトンが表す言語の性質
L = 「bが有限回のみ現れる」言語

## よって
Trace /\ L = emptyset
=> 結果のHFLはvalid
