%PROGRAM
let fb (x : int) = if x > 0 then (event a; app fa (x-1)) else (event b; app fb 5)
let fa (x : int) = if x > 0 then (event a; app fa (x-1)) else (event b; app fb 5)
let app (h : int -> unit) (x : int) = h x
let () = fb 5

%INITIAL
qa

%TRANSITION
(qa, a) -> qa
(qb, a) -> qa
(qa, b) -> qb
(qb, b) -> qb

%PRIORITY
qa -> 0
qb -> 1
